// ==UserScript==
// @name         枝江追番表进度保存脚本
// @namespace    http://tampermonkey.net/
// @version      1.4
// @description  只有同时满足“成员匹配”和“日期匹配”时才显示按钮
// @author       Evelynal(伊芙琳娜)
// @match        *://www.bilibili.com/video/*
// @grant        GM_addStyle
// @run-at       document-end
// @license MIT
// ==/UserScript==
 
(function() {
    'use strict';
 
    GM_addStyle(`
        #asoul-jump-btn {
            margin-left: 10px;
            padding: 4px 14px;
            background-color: #fb7299;
            color: white;
            border: none;
            border-radius: 4px;
            cursor: pointer;
            font-size: 12px;
            font-weight: bold;
            vertical-align: middle;
            z-index: 999;
        }
        #asoul-jump-btn:hover { background-color: #ff85ad; }
    `);
 
    const nameMap = {
        '贝拉kira': 'bella', '贝拉': 'bella',
        '嘉然今天吃什么': 'diana', '嘉然': 'diana',
        '乃琳Queen': 'eileen', '乃琳': 'eileen',
        '心宜不是心仪': 'fiona', '心宜': 'fiona',
        '思诺snow': 'gladys', '思诺': 'gladys'
    };
 
    const upNames = ["贝拉kira", "嘉然今天吃什么", "乃琳Queen", "心宜不是心仪", "思诺snow"];
    const keywords = ["贝拉", "嘉然", "乃琳", "心宜", "思诺"];
 
    // 核心提取逻辑
    function getAsoulFormat() {
        const upNameElement = document.querySelector('.up-name');
        const titleElement = document.querySelector('.video-title');
        if (!upNameElement || !titleElement) return null;
 
        const currentUp = upNameElement.innerText.trim();
        const currentTitle = titleElement.innerText.trim();
 
        // 1. 检测成员匹配 (UP主或标题)
        let matchedKey = "";
        if (upNames.includes(currentUp)) {
            matchedKey = nameMap[currentUp];
        } else {
            const found = keywords.find(k => currentTitle.includes(k));
            if (found) matchedKey = nameMap[found];
        }
 
        // 如果成员都没对上，直接返回 null
        if (!matchedKey) return null;
 
        // 2. 检测日期匹配 (必须包含日期关键词)
        let datePart = "";
        let timePart = "0000";
 
        const reg1 = /(\d{4})年(\d{1,2})月(\d{1,2})日(?:(\d{1,2})点)?/;
        const reg2 = /(\d{4})\.(\d{1,2})\.(\d{1,2})/;
 
        const match1 = currentTitle.match(reg1);
        const match2 = currentTitle.match(reg2);
 
        if (match1) {
            datePart = match1[1] + match1[2].padStart(2, '0') + match1[3].padStart(2, '0');
            if (match1[4]) timePart = match1[4].padStart(2, '0') + "00";
        } else if (match2) {
            datePart = match2[1] + match2[2].padStart(2, '0') + match2[3].padStart(2, '0');
        }
 
        // --- 修改点：如果没有从标题中匹配到日期，直接返回 null，不再使用上传日期兜底 ---
        if (!datePart) return null;
 
        return `${datePart}-${timePart}-${matchedKey}@asoul.love`;
    }
 
    function handleJump() {
        const setLinkValue = getAsoulFormat();
        const video = document.querySelector('video');
        if (!setLinkValue || !video) return;
 
        const currentTime = Math.floor(video.currentTime);
        const baseUrl = window.location.origin + window.location.pathname;
        const jumpLink = `${baseUrl}?p=1&t=${currentTime}`;
 
        const finalUrl = `http://localhost:3000/ASoul-Calendar/?set_link=${encodeURIComponent(setLinkValue)}&link=${encodeURIComponent(jumpLink)}`;
 
        // 执行无 Referer 跳转
        const link = document.createElement('a');
        link.href = finalUrl;
        link.rel = 'noreferrer noopener';
        link.target = '_self';
        document.body.appendChild(link);
        link.click();
    }
 
    function injectBtn() {
        if (document.getElementById('asoul-jump-btn')) return;
 
        // 只有当 getAsoulFormat 同时拿到成员和日期时，才会返回非空字符串
        const info = getAsoulFormat();
        if (info) {
            const titleContainer = document.querySelector('.video-info-title-inner') || document.querySelector('.video-title');
            if (titleContainer) {
                const btn = document.createElement('button');
                btn.id = 'asoul-jump-btn';
                btn.innerText = '关闭并记录进度';
                btn.onclick = (e) => {
                    e.preventDefault();
                    handleJump();
                };
                titleContainer.appendChild(btn);
            }
        }
    }
 
    // 监听路由变化
    let lastUrl = location.href;
    const observer = new MutationObserver(() => {
        if (location.href !== lastUrl) {
            lastUrl = location.href;
            const oldBtn = document.getElementById('asoul-jump-btn');
            if (oldBtn) oldBtn.remove();
        }
        injectBtn();
    });
 
    observer.observe(document.body, { childList: true, subtree: true });
 
    // 初始执行
    injectBtn();
})();
